// Задание 7.5 Напишите функцию, которая принимает массив чисел и возвращает новый массив, содержащий только уникальные числа, отсортированные в порядке возрастания.

// Решение:

function getSortedUniqueNumbers(numbers) {
  const unique = [...new Set(numbers)];        // Удаляем дубликаты
  return unique.sort((a, b) => a - b);          // Сортируем по возрастанию
}
